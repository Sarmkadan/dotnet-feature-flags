using System.ComponentModel.DataAnnotations;

namespace FeatureFlags.Configuration
{
    public static class DependencyInjectionExtensions
    {
        public static IServiceCollection AddFeatureFlagsConfiguration(this IServiceCollection services, IConfiguration configuration)
        {
            // Register options with validation
            services.Configure<DotnetFeatureFlagsOptions>(configuration.GetSection("FeatureFlags"))
                .AddOptionsValidation();

            return services;
        }

        private static IServiceCollection AddOptionsValidation(this IServiceCollection services)
        {
            services.AddSingleton<IValidateOptions<DotnetFeatureFlagsOptions>, FeatureFlagsOptionsValidator>();
            return services;
        }
    }

    public class DotnetFeatureFlagsOptions
    {
        [Required]
        public bool EnableCache { get; set; } = true;

        [Range(1, 1440)]
        public int CacheDurationMinutes { get; set; } = 5;

        [Range(1, 3650)]
        public int AuditLogRetentionDays { get; set; } = 365;

        [Required]
        public bool EnableAuditLogging { get; set; } = true;

        [Range(1, 500)]
        public int MaxRulesPerFlag { get; set; } = 100;

        [Range(1, 100)]
        public int MaxConditionsPerRule { get; set; } = 50;

        [Range(1, 50)]
        public int MaxVariantsPerFlag { get; set; } = 10;

        [Required]
        public bool LogEvaluationDetails { get; set; } = false;

        [Range(1, 100)]
        public int DefaultRolloutPercentage { get; set; } = 50;
    }

    public class FeatureFlagsOptionsValidator : IValidateOptions<DotnetFeatureFlagsOptions>
    {
        public ValidateOptionsResult Validate(string? name, DotnetFeatureFlagsOptions options)
        {
            if (options.CacheDurationMinutes < 1 || options.CacheDurationMinutes > 1440)
                return ValidateOptionsResult.Fail("CacheDurationMinutes must be between 1 and 1440");

            if (options.AuditLogRetentionDays < 1 || options.AuditLogRetentionDays > 3650)
                return ValidateOptionsResult.Fail("AuditLogRetentionDays must be between 1 and 3650");

            if (options.MaxRulesPerFlag < 1 || options.MaxRulesPerFlag > 500)
                return ValidateOptionsResult.Fail("MaxRulesPerFlag must be between 1 and 500");

            if (options.MaxConditionsPerRule < 1 || options.MaxConditionsPerRule > 100)
                return ValidateOptionsResult.Fail("MaxConditionsPerRule must be between 1 and 100");

            if (options.MaxVariantsPerFlag < 1 || options.MaxVariantsPerFlag > 50)
                return ValidateOptionsResult.Fail("MaxVariantsPerFlag must be between 1 and 50");

            if (options.DefaultRolloutPercentage < 1 || options.DefaultRolloutPercentage > 100)
                return ValidateOptionsResult.Fail("DefaultRolloutPercentage must be between 1 and 100");

            return ValidateOptionsResult.Success;
        }
    }
}

# Feature Flags Configuration

## Configuration Options

All configuration is handled through the `appsettings.json` file. Here are the available settings:

### Connection Strings
| Setting | Description | Example |
|---------|-------------|---------|
| `DefaultConnection` | Database connection string for feature flag storage | `Server=your_server;Database=FeatureFlagEngine;User Id=your_user;Password=your_password;` |

### Feature Flags Configuration
| Setting | Description | Valid Range | Default |
|---------|-------------|-------------|--------|
| `EnableCache` | Enable caching of feature flag evaluations | true/false | true |
| `CacheDurationMinutes` | Duration to cache evaluations in minutes | 1-1440 | 5 |
| `AuditLogRetentionDays` | Number of days to retain audit logs | 1-3650 | 365 |
| `EnableAuditLogging` | Enable audit logging for feature flag changes | true/false | true |
| `MaxRulesPerFlag` | Maximum number of rules allowed per feature flag | 1-500 | 100 |
| `MaxConditionsPerRule` | Maximum number of conditions per rule | 1-100 | 50 |
| `MaxVariantsPerFlag` | Maximum number of variants per feature flag | 1-50 | 10 |
| `LogEvaluationDetails` | Enable detailed evaluation logging for debugging | true/false | false |
| `DefaultRolloutPercentage` | Default percentage for feature flag rollouts | 1-100 | 50 |

### Logging Configuration
| Setting | Description | Example |
|---------|-------------|---------|
| `LogLevel.Default` | Default log level for all categories | "Information" |
| `LogLevel.FeatureFlags` | Specific log level for FeatureFlags namespace | "Debug" |

### Security
| Setting | Description | Example |
|---------|-------------|---------|
| `AllowedHosts` | Allowed hosts for the application | "*" (allow all)
